class Solution {
public:
    vector<int> nextGreaterElement(vector<int>& findNums, vector<int>& nums) {
        stack<int> s;
        unordered_map<int, int> um;
        for(int i=nums.size()-1; ~i; i--){// ~i == i>=0;
            int n = nums[i];
            while(!s.empty() && s.top()<n) s.pop();
            um[n] = s.size() ? s.top() : -1;
            s.push(n);
        }
        
        vector<int> r;
        for(auto n: findNums){
            r.push_back(um[n]);
        }
        return r;
            
    }
};

class Solution {
public:
    vector<int> nextGreaterElement(vector<int>& findNums, vector<int>& nums) {
        stack<int> s;
        unordered_map<int, int> um;
        for(auto n: nums){
            while(!s.empty() && s.top()<n){
                um[s.top()] = n;
                s.pop();
            }
            s.push(n);
        }
        
        vector<int> r;
        for(auto n: findNums){
            r.push_back(um.count(n)? um[n]:-1);
        }
        return r;
            
    }
};
